### Webhook Handler Update

**File:** `/app/api/stripe/webhook/route.ts`

```typescript
if (event.type === 'checkout.session.completed') {
  const session = event.data.object;
  
  if (session.metadata.programId === 'barber-apprentice') {
    
    // 1. Create enrollment
    const enrollment = await createEnrollment({
      studentId: userId,
      programId: 'barber-apprentice',
      amount: 4890,
    });
    
    // 2. Transfer $300 to Milady
    await stripe.transfers.create({
      amount: 30000,
      currency: 'usd',
      destination: process.env.MILADY_STRIPE_ACCOUNT_ID,
      metadata: {
        studentEmail: session.customer_email,
        enrollmentId: enrollment.id,
      },
    });
    
    // 3. Notify Milady to create account
    await notifyMilady({
      email: session.customer_email,
      firstName: session.customer_details.name.split(' ')[0],
      lastName: session.customer_details.name.split(' ').slice(1).join(' '),
      programId: 'CIMA-BARBER',
    });
    
    // 4. Send welcome email
    await sendWelcomeEmail({
      to: session.customer_email,
      name: session.customer_details.name,
      enrollmentId: enrollment.id,
      includesMilady: true,
      includesRISE: true,
    });
  }
}
```
