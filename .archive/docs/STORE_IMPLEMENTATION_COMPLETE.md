export async function GET(req: Request, { params }: { params: { purchaseId: string } }) {
  // Verify purchase
  const purchase = await getPurchase(params.purchaseId);
  if (!purchase) return new Response('Not found', { status: 404 });
  
  // Verify user owns purchase
  const user = await getCurrentUser();
  if (purchase.userId !== user.id) return new Response('Unauthorized', { status: 401 });
  
  // Generate download link (expires in 1 hour)
  const downloadUrl = await generateSecureDownloadUrl(purchase.productId);
  
  // Log download
  await logDownload(purchase.id);
  
  // Redirect to download
  return Response.redirect(downloadUrl);
}

export async function cloneAndDeployCodebase(
  productId: string,
  customerId: string,
  licenseKey: string,
  deploymentOptions: {
    platform: 'vercel' | 'netlify' | 'aws' | 'manual',
    domain?: string,
    envVars?: Record<string, string>,
  }
) {
  // 1. Clone repository
  const repo = await cloneRepository(productId);
  
  // 2. Inject license key
  await injectLicense(repo, licenseKey);
  
  // 3. Configure environment
  await configureEnvironment(repo, deploymentOptions.envVars);
  
  // 4. Deploy to platform
  if (deploymentOptions.platform !== 'manual') {
    await deployToplatform(repo, deploymentOptions);
  }
  
  // 5. Send deployment email
  await sendDeploymentEmail(customerId, repo, deploymentOptions);
  
  return {
    repoUrl: repo.url,
    deploymentUrl: deploymentOptions.domain,
    status: 'deployed',
  };
}
