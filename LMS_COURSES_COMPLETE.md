```sql
\i supabase/migrations/20241116_create_lms_courses_part4.sql
\i supabase/migrations/20241116_create_medical_assistant_course.sql
```

### Step 2: Verify Data

```sql
-- Check programs
SELECT slug, title, funding FROM programs ORDER BY title;

-- Check courses
SELECT slug, title, duration_hours, metadata->>'format' as format
FROM courses
ORDER BY title;

-- Check modules
SELECT c.title as course, m.title as module, m.order_index
FROM modules m
JOIN courses c ON c.id = m.course_id
ORDER BY c.title, m.order_index;
```
